class Something{
  int speed=0;
  void accelerate()
  {
    speed+=30;
  }
}
class Vehicle extends Something{
  int speed=10;
  bool isEngineOn=true;
  void accelerate()
  {
    speed+=5;
  }
}
class Car extends Vehicle{
  int noOfWheels=4;
  void printSomething()
  {
    print(noOfWheels);
  }
}

class Truck extends Something{
  int noOfWheels=6;
  int speed=30;
  bool isEngineOn=false;
  
  
}
